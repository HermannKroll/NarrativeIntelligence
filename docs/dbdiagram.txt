# This is the description of the database schema (03/03/2020).
# Code can be pasted on https://app.quickdatabasediagrams.com to generate the diagram

Document
-
id PK bigint
collection PK string
title string
abstract string
fulltext string NULLABLE
date_inserted datetime

DocumentTranslation
-
document_id int PK FK >- Document.id
document_collection string PK  FK >- Document.collection
source_doc_id string
md5 string
date_inserted DateTime
source string

Tagger
-
name PK string
version PK string

DocTaggedBy
-
document_id int PK FK >- Document.id
document_collection string PK  FK >- Document.collection
tagger_name string PK FK >- Tagger.name
tagger_version string PK FK >- Tagger.version
ent_type PK string
date_inserted datetime

Tag
-
id int PK
ent_type string UNIQUE
start int UNIQUE
end int UNIQUE
ent_id string UNIQUE
ent_str string
document_id int UNIQUE FK >- Document.id
document_collection string UNIQUE FK >- Document.collection

Predication
-
id bigint PK
document_id bigint FK >- Document.id
document_collection string FK >- Document.collection
subject_id string
subject_str string
subject_type string
predicate string
predicate_canonicalized string NULLABLE
object_id string
object_str string
object_type string
confidence float NULLABLE
sentence_id bigint FK >- Sentence.id
extraction_type string

Sentence
-
id bigint PK
document_id bigint FK >- Document.id
document_collection string FK >- Document.collection
text string
md5hash string

DocProcessedByIE:
-
document_id bigint FK >- Document.id
document_collection string FK >- Document.collection
extraction_type string
date_inserted DateTime
